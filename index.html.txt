    <script>
        const tg = window.Telegram.WebApp;
        tg.expand(); // Разворачиваем окно на весь экран

        function startLogin() {
            // Открываем окно логина SoundCloud
            const loginWin = window.open("https://soundcloud.com/signin", "_blank", "width=500,height=600");
            
            // Каждые 2 секунды проверяем, появился ли токен в куках
            const checkTimer = setInterval(() => {
                try {
                    // Пытаемся достать токен из хранилища браузера
                    const token = document.cookie.split('; ').find(row => row.startsWith('oauth_token='))?.split('=')[1] 
                                  || localStorage.getItem('oauth_token');

                    if (token) {
                        clearInterval(checkTimer);
                        loginWin.close();
                        // ОТПРАВЛЯЕМ ТОКЕН БОТУ
                        tg.sendData(token); 
                        tg.close();
                    }
                } catch (e) {
                    // Ошибки кросс-домена — это нормально, пока юзер не залогинится
                }
            }, 2000);
        }
    </script>
